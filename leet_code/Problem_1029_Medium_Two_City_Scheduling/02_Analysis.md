public class Solution {
    // Rối bời DP
    public int twoCitySchedCost(int[][] costs) {
        int n = costs.length / 2;
        int[][] dp = new int[n + 1][n + 1];
        
        for (int i = 1; i <= n; i++) dp[i][0] = dp[i - 1][0] + costs[i - 1][0];
        for (int j = 1; j <= n; j++) dp[0][j] = dp[0][j - 1] + costs[j - 1][1];
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = Math.min(dp[i - 1][j] + costs[i + j - 1][0], 
                                    dp[i][j - 1] + costs[i + j - 1][1]);
            }
        }
        return dp[n][n];
    }
}

import java.util.Arrays;

public class Solution {
    public int twoCitySchedCost(int[][] costs) {
        // Cầm thớt băm mảng Sắp Xếp Đòn Đẩy Theo Chỉ Số Lãi Refund: Chênh lệch Bay A Trừ Bay B
        Arrays.sort(costs, (a, b) -> Integer.compare(a[0] - a[1], b[0] - b[1]));

        int totalMinCost = 0;
        int halfN = costs.length / 2;

        // Bốc lột Nửa Đỏ (Phần Âm Thua Thiệt Nặng) Ép Nhảy Xuống Biển Hố A
        // Bốc lốt Nửa Xanh (Nặng Giá Khống Điêu B) Ép Dội Về Vùng An Toàn Bỏ Tổ Lạch
        for (int i = 0; i < halfN; i++) {
            totalMinCost += costs[i][0];
            totalMinCost += costs[i + halfN][1];
        }

        return totalMinCost;
    }
}
